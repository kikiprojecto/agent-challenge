/// Sorts a vector of integers in descending order
///
/// # Examples
/// ```
/// let result = sort_numbers(vec![7, 5, 0, 3, 9, 2, 4, 8, 5, 4]);
/// assert_eq!(result, vec![9, 8, 7, 5, 5, 4, 4, 3, 2, 0]);
/// ```
pub fn sort_numbers(mut numbers: Vec<i32>) -> Vec<i32> {
    numbers.sort_unstable();
    numbers.reverse();
    numbers
}

fn main() {
    let nums = vec![7, 5, 0, 3, 9, 2, 4, 8, 5, 4];
    let sorted = sort_numbers(nums);
    println!("Sorted: {:?}", sorted);
}
